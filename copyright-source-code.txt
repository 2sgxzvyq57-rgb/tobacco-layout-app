function drawArrow(
  ctx: CanvasRenderingContext2D,
  x: number, y: number,
  dir: 'left' | 'right' | 'up' | 'down',
  size: number
) {

      ctx.lineTo(x + size, y - size / 2);
      ctx.lineTo(x + size, y + size / 2);
      break;
    case 'left':
      ctx.moveTo(x, y);
      ctx.lineTo(x - size, y - size / 2);
      ctx.lineTo(x - size, y + size / 2);
      break;
    case 'up':
      ctx.moveTo(x, y);
      ctx.lineTo(x - size / 2, y - size);
      ctx.lineTo(x + size / 2, y - size);
      break;
    case 'down':
      ctx.moveTo(x, y);
      ctx.lineTo(x - size / 2, y + size);
      ctx.lineTo(x + size / 2, y + size);
      break;
  }
  ctx.closePath();
  ctx.fill();
}

  name: string;
  type: ObjectType;

  y: number;        // 左下角Y坐标（米）
  width: number;    // 宽度（米）

  rotation: number; // 旋转角度（0-360度），0度表示默认朝向
}
